import torch
import torchvision
import torchvision.transforms as transforms
import torch.nn as nn
import matplotlib.pyplot as plt
import os

# Transform
transform = transforms.Compose([
    transforms.Resize((224, 224)),
    transforms.ToTensor()
])

dataset = torchvision.datasets.ImageFolder("dataset/train", transform=transform)
loader = torch.utils.data.DataLoader(dataset, batch_size=1, shuffle=True)

# Model (same as yours)
class CNN(nn.Module):
    def __init__(self):
        super().__init__()
        self.conv = nn.Sequential(
            nn.Conv2d(3,16,3),
            nn.ReLU(),
            nn.MaxPool2d(2),
            nn.Conv2d(16,32,3),
            nn.ReLU(),
            nn.MaxPool2d(2)
        )
        self.fc = nn.Sequential(
            nn.Linear(32*54*54,128),
            nn.ReLU(),
            nn.Linear(128,2)
        )

    def forward(self,x):
        x = self.conv(x)
        x = x.view(x.size(0), -1)
        return self.fc(x)

model = CNN()
model.load_state_dict(torch.load("model.pth"))
model.eval()

# DeepFool Attack
def deepfool(image, model, num_classes=2, overshoot=0.02, max_iter=50):
    image = image.clone().detach().requires_grad_(True)
    output = model(image)
    label = output.argmax()

    perturbed = image.clone()

    for _ in range(max_iter):
        output = model(perturbed)
        if output.argmax() != label:
            break

        loss = output[0, label]
        model.zero_grad()
        loss.backward(retain_graph=True)

        grad = perturbed.grad.data
        perturbed = perturbed + overshoot * grad.sign()
        perturbed = perturbed.detach().requires_grad_(True)

    return perturbed

# Save images
save_path = "detector_dataset/adversarial/"
os.makedirs(save_path, exist_ok=True)

count = 0

for data, target in loader:
    adv = deepfool(data, model)

    img = adv.detach().numpy().transpose(1,2,0)
    img = img.clip(0,1)

    plt.imsave(save_path + f"deepfool_{count}.png", img)
    count += 1

    if count >= 100:
        break

print("DeepFool images generated")
